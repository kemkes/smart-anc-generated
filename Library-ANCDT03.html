using FHIR version '4.0.1'

include FHIRHelpers version '4.0.1'

include ANCConcepts called Cx
include ANCDataElements called PatientData

/*
&quot;Current physiological symptoms&quot; = &quot;Nausea and vomiting&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message)&quot;:
  ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Nausea and Vomiting Choices&quot;

/*
&quot;Persistent physiological symptoms&quot; = &quot;Nausea and vomiting&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 2&quot;:
  ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Nausea and Vomiting Choices&quot;

/*
&quot;Current physiological symptoms&quot; = &quot;Heartburn&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 3&quot;:
  ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Heartburn Choices&quot;

/*
&quot;Persistent physiological symptoms&quot; = &quot;Heartburn&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 4&quot;:
  ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Heartburn Choices&quot;

/*
&quot;Current physiological symptoms&quot; = &quot;Leg cramps&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 5&quot;:
  ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Leg cramps Choices&quot;

/*
&quot;Persistent physiological symptoms&quot; = &quot;Leg cramps&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 6&quot;:
  ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Leg cramps Choices&quot;

/*
(((&quot;Current physiological symptoms&quot; = &quot;Low back pain&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Contractions&quot; OR &quot;Pain during urination (dysuria)&quot; OR &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;))
  OR ((&quot;Current physiological symptoms&quot; = &quot;Pelvic pain&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Contractions&quot; OR &quot;Pain during urination (dysuria)&quot; OR &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;))
  OR ((&quot;Persistent physiological symptoms&quot; = &quot;Low back pain&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Contractions&quot; OR &quot;Pain during urination (dysuria)&quot; OR &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;)))
  OR ((&quot;Persistent physiological symptoms&quot; = &quot;Pelvic pain&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Contractions&quot; OR &quot;Pain during urination (dysuria)&quot; OR &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;))
*/
define &quot;Should Conduct counselling on physiological symptoms&quot;:
  (
    ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Low back Choices&quot;
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Contractions Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - during urination (dysuria) Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Extreme pelvic pain Choices&quot;)
  ) or
  (
    ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Pelvic Choices&quot;
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Contractions Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - during urination (dysuria) Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Extreme pelvic pain Choices&quot;)
  ) or
  (
    ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Pain - Low back Choices&quot;
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Contractions Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - during urination (dysuria) Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Extreme pelvic pain Choices&quot;)
  ) or
  (
    ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Pain - Pelvic Choices&quot;
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Contractions Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - during urination (dysuria) Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Extreme pelvic pain Choices&quot;)
  )

/*
&quot;Current physiological symptoms&quot; = &quot;Constipation&quot;
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 7&quot;:

((((&quot;Current physiological symptoms&quot; = &quot;Varicose veins&quot;)
    AND (Current physiological symptoms&quot; ≠ &quot;Leg pain&quot; OR &quot;Leg redness&quot;))
  OR ((&quot;Current physiological symptoms&quot; = &quot;Oedema&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Leg pain&quot; OR &quot;Leg redness&quot;)))
  OR ((&quot;Persistent physiological symptoms&quot; = &quot;Varicose veins&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Leg pain&quot; OR &quot;Leg redness&quot;)))
  OR ((&quot;Persistent physiological symptoms&quot; = &quot;Oedema&quot;)
  AND (Current physiological symptoms&quot; ≠ &quot;Leg pain&quot; OR &quot;Leg redness&quot;))
*/
define &quot;Should Conduct counselling on physiological symptoms (see Annotations for pop-up message) 9&quot;:
  (
    (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Varicose veins Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Leg Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Leg redness Choices&quot;)
  )
  or (
    (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Oedema Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Leg Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Leg redness Choices&quot;)
  )
  or (
    (ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Varicose veins Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Leg Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Leg redness Choices&quot;)
  )
  or (
    (ContactData.&quot;Persistent physiological symptoms&quot; in Cx.&quot;Persistent physiological symptoms - Oedema Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Pain - Leg Choices&quot;)
      and not (ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Leg redness Choices&quot;)
  )

/*
(((&quot;Current physiological symptoms&quot; = &quot;Contractions&quot;)
  OR (Current physiological symptoms&quot; = &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;))
  OR (&quot;Persistent physiological symptoms&quot; = &quot;Contractions&quot;))
  OR (Persistent physiological symptoms&quot; = &quot;Extreme pelvic pain, cannot walk (symphysis pubis dysfunction)&quot;)

  ContactData.&quot;Current physiological symptoms&quot; in Cx.&quot;Current physiological symptoms - Contractions Choices&quot;
